📄 digital_twin.m — Kalman filter kern
%% Initialisatie
twin_config;
cantoni_LMI;                 % laadt comb_Pool_disc vanuit WIS-sim
A = comb_Pool_disc.A;
B = comb_Pool_disc.B;
C = comb_Pool_disc.C;

x_hat = zeros(6,1);
P     = eye(6);

%% Per tijdstap
function twin_update(y_meas, u, epoch)
  x_prior   = A * x_hat + B * u;
  P_prior   = A * P * A' + Q;
  innovatie = y_meas - C * x_prior;
  K         = P_prior * C' / (C * P_prior * C' + R);
  x_hat     = x_prior + K * innovatie;
  P         = (eye(6) - K*C) * P_prior;

  twin_log(epoch, y_meas, C*x_prior, innovatie, u);
  if PLOT_LIVE; twin_plot(); end
end
