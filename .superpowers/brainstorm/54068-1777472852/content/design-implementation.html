📄 digital_twin.m — Kalman filter kern
%% Initialisatie (eenmalig)
twin_config;
cantoni_LMI;                 % laadt comb_Pool_disc
A = comb_Pool_disc.A;
B = comb_Pool_disc.B;
C = comb_Pool_disc.C;

x_hat = zeros(6,1);         % initiële staatsschatting
P     = eye(6);             % initiële foutcovariantie

%% Per tijdstap (callback of loop)
function twin_update(y_meas, u)

  % — Predict —
  x_prior = A * x_hat + B * u;
  P_prior = A * P * A' + Q;

  % — Innovatie (afwijking vóór correctie) —
  innovatie = y_meas - C * x_prior;

  % — Kalman gain —
  K = P_prior * C' / (C * P_prior * C' + R);

  % — Update —
  x_hat = x_prior + K * innovatie;
  P     = (eye(6) - K*C) * P_prior;

  % — Log & visualiseer —
  twin_log(epoch, y_meas, C*x_prior, innovatie, u);
  if PLOT_LIVE; twin_plot(); end
end
